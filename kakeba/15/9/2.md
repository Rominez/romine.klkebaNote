```yml
spring:
  cloud:
    nacos:
      discovery:
        server-addr: localhost:8881
    sentinel:
      transport:
        # sentinel 注册地址
        dashboard: localhost:8080
        # 服务接收 sentinel 的配置端口
        port: 8888
        # 该配置能够使 dashboard 主动发现该应用
        # 如果不配置，调用 api 时才会出现在 sentinel 中  
        eager: true
```
