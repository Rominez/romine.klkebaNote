```yml
spring:
  cloud:
    stream:
      kafka:
        binder:
          # 配置集群地址，多个用逗号隔开
          brokers: KAFKAIP:PORT
          # 自动创建主题
          auto-create-topics: true
        bindings:
          # 绑定的管道
          # output 对应的是 MessageChannel 自动注入时绑定的 BeanName
          output:
            # 指定管道中的消息的主题，这个主题会在 kafka 中自动创建
            destination: TOPIC1
            content-type: text/plain
          # 输入管道格式相同，不用指定类型
          input:
            destination: TOPIC1
```
