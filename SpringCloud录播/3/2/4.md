```yml
feign:
  compression:
    request:
      # 启用 feign 客户端发出的请求数据压缩
      enabled: true
      # 配置需要进行压缩的数据格式  
      mime-type: text/xml,application/xml,application/json
      # 配置需要压缩的数据量的最小值
      min-request-size: 2048
    response:
      # 启用 feign 客户端的响应数据压缩
      enabled: true
  client:
    config:
      default:
        # feign 连接提供者的超时时限  
        connectTimeout: 5000
        # feign 连接上提供者后，从提交请求到收到响应的超时时限  
        readTimeout: 5000

```
